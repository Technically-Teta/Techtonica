// Write the function isEmpty(obj) which returns true if the object has no properties, false otherwise.
//Explanation:

// The isEmpty function takes an object obj as a parameter.
// It iterates over the keys in the object using a for...in loop.
// Inside the loop, it checks if the object has the property key using hasOwnProperty.
// If a property is found, it immediately returns false because the object is not empty.
// If no properties are found, it continues to the next iteration.
// Finally, if the loop completes without returning false, it means the object is empty, and the function returns true.
// In your example, schedule is an empty object, so isEmpty(schedule) will return true. After adding a property to schedule, isEmpty(schedule) will 
return false because the object is no longer empty.

// TASK 2 
  function isEmpty(obj) {
  for (let key in obj) {
    if (obj.hasOwnProperty(key)) {
      return false;
    }
  }
  return true;
}

let schedule = {};

console.log(isEmpty(schedule)); // true

schedule["8:30"] = "get up";

console.log(isEmpty(schedule)); // false

function multiplyNumeric(obj) {
  for (let key in obj) {
    if (typeof obj[key] == 'number') {
      obj[key] *= 2;
    }
  }
}
